import requests
from bs4 import BeautifulSoup
import os
import re

BASE_URL = "https://www.gov.br"
URL = "https://www.gov.br/anvisa/pt-br/centraisdeconteudo/publicacoes/servicosdesaude/manuais/cadernos-de-seguranca-do-paciente-e-qualidade-em-servicos-de-saude-2024-versoes-preliminares-nao-finalizadas-aguardando-o-envio-de-sugestoes"

headers = {
    "User-Agent": "Mozilla/5.0"
}

output_dir = "cadernos"
os.makedirs(output_dir, exist_ok=True)

print("🔎 Buscando cadernos...")

res = requests.get(URL, headers=headers)
soup = BeautifulSoup(res.text, "html.parser")

# pegar links dos cadernos
caderno_links = []

for a in soup.find_all("a", href=True):
    if "caderno-" in a["href"]:
        link = a["href"]

        if link.startswith("/"):
            link = BASE_URL + link

        if link not in caderno_links:
            caderno_links.append(link)

print(f"📚 Encontrados {len(caderno_links)} cadernos")

# acessar cada caderno
for caderno_url in caderno_links:
    print(f"\n🔎 Acessando: {caderno_url}")

    res = requests.get(caderno_url, headers=headers)
    soup = BeautifulSoup(res.text, "html.parser")

    download_link = None

    for a in soup.find_all("a", href=True):
        if "@@download" in a["href"]:
            download_link = a["href"]
            break

    if not download_link:
        print("❌ PDF não encontrado")
        continue

    if download_link.startswith("/"):
        download_link = BASE_URL + download_link

    print(f"⬇️ Baixando: {download_link}")

    pdf = requests.get(download_link, headers=headers)

    # ===== PEGAR NOME DO ARQUIVO =====
    filename = None

    # 1. tentar pelo header
    if "content-disposition" in pdf.headers:
        match = re.findall(
            'filename="(.+)"',
            pdf.headers["content-disposition"]
        )
        if match:
            filename = match[0]

    # 2. tentar pela URL
    if not filename:
        filename = os.path.basename(download_link)
        if not filename.endswith(".pdf"):
            filename = None

    # 3. fallback: título da página
    if not filename:
        title = soup.title.string.strip()
        filename = title.replace(" ", "_") + ".pdf"

    # limpar caracteres inválidos
    filename = filename.replace("/", "_") \
                       .replace("\\", "_") \
                       .replace(":", "_") \
                       .replace("%20", " ")

    filepath = os.path.join(output_dir, filename)

    with open(filepath, "wb") as f:
        f.write(pdf.content)

    print(f"✅ Salvo: {filename}")

print("\n🎉 FINALIZADO")
